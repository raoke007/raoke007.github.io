Merge two sorted linked lists and return it as a new list. The new list should be made by splicing together the nodes of the first two lists.

Example:
  Input: 1->2->4, 1->3->4
  Output: 1->1->2->3->4->4

暴力实现法

思路：

  • 使用List存储两个链表数据
  • 对List进行升序排序
  • 遍历List，生成链表返回

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        if (l1 == null && l2 == null) {
            return null;
        }
        
        List<Integer> mergeList = new ArrayList();
        
        while (l1 != null) {
            mergeList.add(l1.val);
            l1 = l1.next;
        }
        
        while (l2 != null) {
            mergeList.add(l2.val);
            l2 = l2.next;
        }
        
        Collections.sort(mergeList);
        
        ListNode root = new ListNode(mergeList.get(0));
        ListNode cur = root;
        for (int i = 1; i < mergeList.size(); i++) {
            cur.next = new ListNode(mergeList.get(i));
            cur = cur.next;
        }
        
        return root;
    }
}

优化

思路：

  • 上述解决方案，执行效率过低，依次遍历链表 --> 集合存储数值 --> 升序排序集合 --> 遍历集合重新生成链表

  • 分别获取2个链表的长度，以元素多（以链表l1为主举例）的链表为主进行递归调用

  • 每次递归调用的逻辑为：取l1、l2中值最小的元素值（例如此时取l1的值）为节点，继续递归时传入l1.next和l2

  • 递归调用的终止条件：如果l1节点为空，则返回l2（此时l2可能还有值）

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
class Solution {
    public ListNode mergeTwoLists(ListNode l1, ListNode l2) {
        ListNode newL1 = l1;
        ListNode newL2 = l2;
        // 分别获取链表的长度
        int len1 = 0;
        while (l1 != null) {
            len1++;
            l1 = l1.next;
        }

        int len2 = 0;
        while (l2 != null) {
            len2++;
            l2 = l2.next;
        }

        // 以链表长的为主进行递归调用
        if (len1 >= len2) {
            return addTwoNumbers(newL1, newL2);
        } else {
            return addTwoNumbers(newL2, newL1);
        }
    }
    
    private ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        if (l1 == null) {
            return l2;
        }

        ListNode node;
        if (l2 != null) {
            // 取最小值当做节点
            if (l1.val <= l2.val) {
                node = new ListNode(l1.val);
                // 此时使用l1.val，则下一次递归过程传入l1.next和l2
                node.next = addTwoNumbers(l1.next, l2);
            } else {
                node = new ListNode(l2.val);

                if (l2.next != null) {
                    // 此时使用l2.val，则下一次递归过程传入l1和l2.next
                    node.next = addTwoNumbers(l1, l2.next);
                } else {
                    node.next = addTwoNumbers(l1, null);
                }
            }
        } else {
            node = new ListNode(l1.val);
            node.next = addTwoNumbers(l1.next, null);
        }

        return node;
    }
}
